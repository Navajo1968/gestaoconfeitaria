-- Ajustando o script para compatibilidade com PostgreSQL

CREATE TABLE clientes (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL
);

CREATE TABLE pedidos (
    id SERIAL PRIMARY KEY,
    cliente_id INT NOT NULL,
    data_pedido TIMESTAMP NOT NULL,
    FOREIGN KEY (cliente_id) REFERENCES clientes(id)
);

-- Removendo referências "IDENTITY" e "GO"
-- Substituindo chaves primárias "CLUSTERED" por comandos compatíveis com PostgreSQL
-- Garantindo que FOREIGN KEYS utilizem a sintaxe e o funcionamento do PostgreSQL
